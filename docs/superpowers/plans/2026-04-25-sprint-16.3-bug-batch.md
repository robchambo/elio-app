/// Cycle a single item's tier on tap.
/// Staples: alwaysHave → almostAlwaysHave → removed.
/// Perishables: fresh → thisWeek → today → removed.
void _cycleItemTier(Map<String, dynamic> item) {
  setState(() {
    final tier = item['tier'] as String?;
    final isPerishable = tier == 'perishable' ||
        item['expiryDate'] != null; // perishables carry expiryDate
    if (isPerishable) {
      // Cycle by expiry bucket (today/thisWeek/fresh) → remove
      // Implementation matches screen 12 cycle; consult that file for
      // exact bucket-to-expiryDate mapping.
      _cyclePerishableBucket(item);
    } else {
      _cycleStapleTier(item);
    }
  });
}

void _cycleStapleTier(Map<String, dynamic> item) {
  final cur = item['tier'] as String?;
  if (cur == 'alwaysHave') {
    item['tier'] = 'almostAlwaysHave';
  } else {
    // 'almostAlwaysHave' or anything else → remove
    _items.removeWhere((it) => identical(it, item));
  }
}

void _cyclePerishableBucket(Map<String, dynamic> item) {
  // Read current bucket from expiryDate delta.
  final expiryStr = item['expiryDate'] as String?;
  final today = DateTime.now();
  if (expiryStr == null) {
    // 'fresh' default → move to thisWeek (5 days out)
    item['expiryDate'] = today.add(const Duration(days: 5)).toIso8601String();
    return;
  }
  final exp = DateTime.tryParse(expiryStr);
  if (exp == null) {
    _items.removeWhere((it) => identical(it, item));
    return;
  }
  final days = exp.difference(today).inDays;
  if (days > 6) {
    // fresh → thisWeek
    item['expiryDate'] = today.add(const Duration(days: 5)).toIso8601String();
  } else if (days > 0) {
    // thisWeek → today
    item['expiryDate'] = today.toIso8601String();
  } else {
    // today → remove
    _items.removeWhere((it) => identical(it, item));
  }
}

        final exp = DateTime(expiry.year, expiry.month, expiry.day);
        final days = exp.difference(today).inDays;
